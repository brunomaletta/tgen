57// Returns a random number in [l, r].
58template <typename T> T next(T l, T r) {
59 tgen_ensure(l <= r, "range for `next` bust be valid");
60 if constexpr (std::is_integral_v<T>)
61 return std::uniform_int_distribution<T>(l, r)(rng_internal);
62 else if constexpr (std::is_floating_point_v<T>)
63 return std::uniform_real_distribution<T>(l, r)(rng_internal);
64 else
65 throw error_internal("invalid type for next (" +
66 std::string(typeid(T).name()) + ")");
67}

69// Shuffles [first, last) inplace uniformly.
70template <typename It> void shuffle(It first, It last) {
71 if (first == last)
72 return;
73
74 for (It i = first + 1; i != last; ++i)
75 std::iter_swap(i, first + next(0, static_cast<int>(i - first)));
76}
77
78// Shuffles container uniformly.
79template <typename C> [[nodiscard]] C shuffle(const C &container) {
80 auto new_container = container;
81 shuffle(new_container.begin(), new_container.end());
82 return new_container;
83}

85// Returns a random element from [first, last).
86template <typename It> typename It::value_type any(It first, It last) {
87 int size = std::distance(first, last);
88 It it = first;
89 std::advance(it, next(0, size - 1));
90 return *it;
91}

101// Chooses k values from the container, as in a subsequence of size k. Returns a
102// copy.
103template <typename C> C choose(int k, const C &container) {
104 tgen_ensure(0 < k and k <= container.size(),
105 "number of elements to choose must be valid");
106 std::vector<typename C::value_type> new_vec;
107 C new_container;
108 int need = k, left = container.size();
109 for (auto cur_it = container.begin(); cur_it != container.end(); ++cur_it) {
110 if (next(1, left--) <= need) {
111 new_container.insert(new_container.end(), *cur_it);
112 need--;
113 }
114 }
115 return new_container;
116}
117template <typename T>
118std::vector<T> choose(int k, const std::initializer_list<T> &il) {
119 return choose(k, std::vector<T>(il.begin(), il.end()));
120}

122// Base struct for generators.
123template <typename GEN> struct gen_base {
124 // Calls the generator until predicate is true.
125 template <typename PRED, typename... Args>
126 auto gen_until(PRED predicate, int max_tries, Args &&...args) {
127 for (int i = 0; i < max_tries; ++i) {
128 auto inst =
129 static_cast<GEN *>(this)->gen(std::forward<Args>(args)...);
130
131 if (predicate(inst))
132 return inst;
133 }
134
135 throw error_internal("could not generate instance matching predicate");
136 }
137 template <typename PRED, typename T, typename... Args>
138 auto gen_until(PRED predicate, int max_tries, std::initializer_list<T> il,
139 Args &&...args) {
140 return gen_until(predicate, max_tries, std::vector<T>(il),
141 std::forward<Args>(args)...);
142 }

145 friend std::ostream &operator<<(std::ostream &out, const gen_base &) {
146 static_assert(
147 false,
148 "you cannot print a generator. Maybe you forgot to call `gen()`?");
149 return out;
150 }

159/*
160 * Opts - options given to the generator.
161 *
162 * Incompatible with testlib.
163 *
164 * Opts are a list of either positional or named options.
165 *
166 * Named options is given in one of the following formats:
167 * 1) -keyname=value or --keyname=value (ex. -n=10 , --test-count=20)
168 * 2) -keyname value or --keyname value (ex. -n 10 , --test-count 20)
169 *
170 * Positional options are number from 0 sequentially.
171 * For example, for "10 -n=20 str" positional option 1 is the string "str".
172 */
173
174inline std::vector<std::string>
175 pos_opts_internal; // Dictionary containing the positional parsed opts.
176inline std::map<std::string, std::string>
177 named_opts_internal; // Global dictionary the named parsed opts.
178
179// Returns true if there is an opt at a given index.
180inline bool has_opt(std::size_t index) {
181 return 0 <= index and index < pos_opts_internal.size();
182}
183
184// Returns true if there is an opt with a given key.
185inline bool has_opt(const std::string &key) {
186 return named_opts_internal.count(key) != 0;
187}
188
189template <typename T> T get_opt_internal(const std::string &value) {
190 try {
191 if constexpr (std::is_same_v<T, bool>) {
192 if (value == "true" or value == "1")
193 return true;
194 if (value == "false" or value == "0")
195 return false;
196 } else if constexpr (std::is_integral_v<T>) {
197 if constexpr (std::is_unsigned_v<T>)
198 return static_cast<T>(std::stoull(value));
199 else
200 return static_cast<T>(std::stoll(value));
201 } else if constexpr (std::is_floating_point_v<T>)
202 return static_cast<T>(std::stold(value));
203 else
204 return value; // default: std::string
205 } catch (...) {
206 }
207
208 throw error_internal("invalid value `" + value + "` for type " +
209 typeid(T).name());
210}
211
212// Returns the parsed opt by a given key. If no opts with the given key are
213// found, returns the given default_value.
214template <typename T, typename KEY>
215T opt(const KEY &key, std::optional<T> default_value = std::nullopt) {
216 if constexpr (std::is_same_v<KEY, int>) {
217 if (!has_opt(key)) {
218 if (default_value)
219 return *default_value;
220 throw error_internal("cannot find key with index " +
221 std::to_string(key));
222 }
223 return get_opt_internal<T>(pos_opts_internal[key]);
224 } else { // std::string
225 if (!has_opt(key)) {
226 if (default_value)
227 return *default_value;
228 throw error_internal("cannot find key with key " +
229 std::string(key));
230 }
231 return get_opt_internal<T>(named_opts_internal[key]);
232 }
233}
234
235inline void parse_opts_internal(int argc, char **argv) {
236 // Parses the opts into `pos_opts_internal` vector and `named_opts_internal`
237 // map. Starting from 1 to ignore the name of the executable.
238 for (int i = 1; i < argc; i++) {
239 std::string key(argv[i]);
240
241 if (key[0] == '-') {
242 tgen_ensure(key.size() > 1,
243 "invalid opt (" + std::string(argv[i]) + ")");
244 if ('0' <= key[1] and key[1] <= '9') {
245 // This case is a positional negative number argument
246 pos_opts_internal.push_back(key);
247 continue;
248 }
249
250 // pops first char '-'
251 key = key.substr(1);
252 } else {
253 // This case is a positional argument that does not start with '-'
254 pos_opts_internal.push_back(key);
255 continue;
256 }
257
258 // Pops a possible second char '-'.
259 if (key[0] == '-') {
260 tgen_ensure(key.size() > 1,
261 "invalid opt (" + std::string(argv[i]) + ")");
262
263 // pops first char '-'
264 key = key.substr(1);
265 }
266
267 // Assumes that, if it starts with '-' and second char is not a digit,
268 // then it is a <key, value> pair.
269 // 1 or 2 chars '-' have already been poped.
270
271 std::size_t eq = key.find('=');
272 if (eq != std::string::npos) {
273 // This is the '--key=value' case.
274 std::string value = key.substr(eq + 1);
275 key = key.substr(0, eq);
276 tgen_ensure(!key.empty() and !value.empty(),
277 "expected non-empty key/value in opt (" +
278 std::string(argv[1]));
279 tgen_ensure(named_opts_internal.count(key) == 0,
280 "cannot have repeated keys");
281 named_opts_internal[key] = value;
282 } else {
283 // This is the '--key value' case.
284 tgen_ensure(named_opts_internal.count(key) == 0,
285 "cannot have repeated keys");
286 tgen_ensure(argv[i + 1], "value cannot be empty");
287 named_opts_internal[key] = std::string(argv[i + 1]);
288 i++;
289 }
290 }
291}
292inline void set_seed_internal(int argc, char **argv) {
293 std::vector<uint32_t> seed;
294
295 // Starting from 1 to ignore the name of the executable.
296 for (int i = 1; i < argc; ++i) {
297 // We append the number of chars, and then the list of chars.
298 int size_pos = seed.size();
299 seed.push_back(0);
300 for (char *s = argv[i]; *s != '\0'; ++s) {
301 ++seed[size_pos];
302 seed.push_back(*s);
303 }
304 }
305 std::seed_seq seq(seed.begin(), seed.end());
306 rng_internal.seed(seq);
307}
308
309// Registers generator by initializing rnd and parsing opts.
310inline void register_gen(int argc, char **argv) {
311 set_seed_internal(argc, argv);
312
313 pos_opts_internal.clear();
314 named_opts_internal.clear();
315 parse_opts_internal(argc, argv);
316}

328template <typename T> struct sequence : gen_base<sequence<T>> {
329 int size_; // Size of sequence.
330 T value_l_, value_r_; // Range of defined values.
331 std::set<T> values_; // Set of values. If empty, use range. if not,
332 // represents the possible values, and the range
333 // represents the index in this set)
334 std::map<T, int>
335 value_idx_in_set_; // Index of every value in the set above.
336 std::vector<std::pair<T, T>> val_range_; // Range of values of each index.
337 std::vector<std::vector<int>> neigh_; // Adjacency list of equality.
338 std::vector<std::set<int>>
339 distinct_constraints_; // All distinct constraints.
340
341 // Creates generator for sequences of size 'size', with random T in [l, r].
342 sequence(int size, T value_l, T value_r)
343 : size_(size), value_l_(value_l), value_r_(value_r), neigh_(size) {
344 tgen_ensure(size_ > 0, "size must be positive");
345 tgen_ensure(value_l_ <= value_r_, "value range must be valid");
346 for (int i = 0; i < size_; ++i)
347 val_range_.emplace_back(value_l_, value_r_);
348 }
349
350 // Creates sequence with value set.
351 sequence(int size, std::set<T> values)
352 : size_(size), values_(values), neigh_(size) {
353 tgen_ensure(size_ > 0, "size must be positive");
354 tgen_ensure(!values.empty(), "value set must be non-empty");
355 value_l_ = 0, value_r_ = values.size() - 1;
356 for (int i = 0; i < size_; ++i)
357 val_range_.emplace_back(value_l_, value_r_);
358 int idx = 0;
359 for (T value : values_)
360 value_idx_in_set_[value] = idx++;
361 }
362
363 // Restricts sequences for sequence[idx] = value.
364 sequence &set(int idx, T value) {
365 tgen_ensure(0 <= idx and idx < size_, "index must be valid");
366 if (values_.size() == 0) {
367 auto &[left, right] = val_range_[idx];
368 if (left == right and value_l_ != value_r_) {
369 tgen_ensure(left == value,
370 "must not set to two different values");
371 } else {
372 tgen_ensure(left <= value and value <= right,
373 "value must be in the defined range");
374 }
375 left = right = value;
376 } else {
377 tgen_ensure(values_.count(value),
378 "value must be in the set of values");
379 auto &[left, right] = val_range_[idx];
380 int new_val = value_idx_in_set_[value];
381 tgen_ensure(left <= new_val and new_val <= right,
382 "must not set to two different values");
383 left = right = new_val;
384 }
385 return *this;
386 }
387
388 // Restricts sequences for sequence[idx_1] = sequence[idx_2].
389 sequence &equal(int idx_1, int idx_2) {
390 tgen_ensure(0 <= std::min(idx_1, idx_2) and
391 std::max(idx_1, idx_2) < size_,
392 "indices must be valid");
393 if (idx_1 == idx_2)
394 return *this;
395
396 neigh_[idx_1].push_back(idx_2);
397 neigh_[idx_2].push_back(idx_1);
398 return *this;
399 }
400
401 // Restricts sequences for sequence[left..right] to have all equal values.
402 sequence &equal_range(int left, int right) {
403 tgen_ensure(0 <= left and left <= right and right < size_,
404 "range indices bust be valid");
405 for (int i = left; i < right; ++i)
406 equal(i, i + 1);
407 return *this;
408 }
409
410 // Restricts sequences for sequence[S] to be distinct, for given subset S of
411 // indices.
412 // You can not add two of these restrictions with intersection.
413 sequence &distinct(std::set<int> indices) {
414 if (!indices.empty())
415 distinct_constraints_.push_back(indices);
416 return *this;
417 }
418
419 // Restricts sequences for sequence[idx_1] != sequence[idx_2].
420 sequence &different(int idx_1, int idx_2) {
421 std::set<int> indices = {idx_1, idx_2};
422 return distinct(indices);
423 }
424
425 // Restricts sequences with distinct elements.
426 sequence &distinct() {
427 std::set<int> indices;
428 for (int i = 0; i < size_; ++i)
429 indices.insert(i);
430 return distinct(indices);
431 }

485 // Generates a uniformly random list of k distinct values in `[value_l,
486 // value_r]`, such that no value is in `forbidden_values`.
487 std::vector<T>
488 generate_distinct_values(int k, const std::set<T> &forbidden_values) {
489 for (auto forbidden : forbidden_values)
490 tgen_ensure(value_l_ <= forbidden and forbidden <= value_r_);
491 // We generate our numbers in the range [0, num_available) with
492 // num_available = (r-l+1)-(forbidden_values.size()), and then map them
493 // to the correct range. We will run k steps of Fisherâ€“Yates, using a
494 // map to store a virtual sequence that starts with a[i] = i.
495 T num_available = (value_r_ - value_l_ + 1) - forbidden_values.size();
496 if (num_available < k)
497 throw error_internal(
498 "failed to generate sequence: complex constraints");
499 std::map<T, T> virtual_list;
500 std::vector<T> gen_list;
501 for (int i = 0; i < k; i++) {
502 T j = next<T>(i, num_available - 1);
503 T vj = virtual_list.count(j) ? virtual_list[j] : j;
504 T vi = virtual_list.count(i) ? virtual_list[i] : i;
505
506 virtual_list[j] = vi, virtual_list[i] = vj;
507
508 gen_list.push_back(virtual_list[i]);
509 }
510
511 // Shifts back to correct range, but there might still be values
512 // that we can not use.
513 for (T &value : gen_list)
514 value += value_l_;
515
516 // Now for every generated value, we shift it by how many forbidden
517 // values are <= to it.
518 std::vector<std::pair<T, int>> values_sorted;
519 for (std::size_t i = 0; i < gen_list.size(); ++i)
520 values_sorted.emplace_back(gen_list[i], i);
521 // We iterate through them in increasing order.
522 std::sort(values_sorted.begin(), values_sorted.end());
523 auto cur_it = forbidden_values.begin();
524 int smaller_forbidden_count = 0;
525 for (auto [val, idx] : values_sorted) {
526 while (cur_it != forbidden_values.end() and
527 *cur_it <= val + smaller_forbidden_count)
528 ++cur_it, ++smaller_forbidden_count;
529 gen_list[idx] += smaller_forbidden_count;
530 }
531
532 return gen_list;
533 }

535 // Generates sequence instance.
536 instance gen() {
537 std::vector<T> vec(size_);
538 std::vector<bool> defined_idx(
539 size_, false); // For every index, if it has been set in `vec`.
540
541 std::vector<int> comp_id(size_, -1); // Component id of each index.
542 std::vector<std::vector<int>> comp(size_); // Component of each comp-id.
543 int comp_count = 0; // Number of different components.
544
545 // Defines value of entire component.
546 auto define_comp = [&](int cur_comp, T val) {
547 for (int idx : comp[cur_comp]) {
548 tgen_ensure(!defined_idx[idx]);
549 vec[idx] = val;
550 defined_idx[idx] = true;
551 }
552 };
553
554 // Groups = components.
555 {
556 std::vector<bool> vis(size_, false); // Visited for each index.
557 for (int idx = 0; idx < size_; ++idx)
558 if (!vis[idx]) {
559 T new_value;
560 bool value_defined = false;
561
562 // BFS to visit the connected component, grouping equal
563 // values.
564 std::queue<int> q({idx});
565 vis[idx] = true;
566 std::vector<int> component;
567 while (!q.empty()) {
568 int cur_idx = q.front();
569 q.pop();
570
571 component.push_back(cur_idx);
572
573 // Checks value.
574 auto [l, r] = val_range_[cur_idx];
575 if (l == r) {
576 if (!value_defined) {
577 // We found the value.
578 value_defined = true;
579 new_value = l;
580 } else if (new_value != l) {
581 // We found a contradiction
582 contradiction_error_internal(
583 "sequence",
584 "tried to set value to `" +
585 std::to_string(new_value) +
586 "`, but it was already set as `" +
587 std::to_string(l) + "`");
588 }
589 }
590
591 for (int nxt_idx : neigh_[cur_idx]) {
592 if (!vis[nxt_idx]) {
593 vis[nxt_idx] = true;
594 q.push(nxt_idx);
595 }
596 }
597 }
598
599 // Group entire component, checking if value is defined.
600 for (int cur_idx : component) {
601 comp_id[cur_idx] = comp_count;
602 comp[comp_id[cur_idx]].push_back(cur_idx);
603 }
604
605 // Sets value if needed.
606 if (value_defined)
607 define_comp(comp_count, new_value);
608
609 ++comp_count;
610 }
611 }
612
613 // Initial parsing of distinct constraints.
614 std::vector<std::set<int>> distinct_containing_comp_idx(comp_count);
615 {
616 int dist_id = 0;
617 for (const std::set<int> &distinct : distinct_constraints_) {
618 // Checks if there are too many distinct values.
619 if (static_cast<unsigned long long>(distinct.size() - 1) +
620 static_cast<unsigned long long>(value_l_) >
621 static_cast<unsigned long long>(value_r_))
622 contradiction_error_internal(
623 "sequence",
624 "tried to generate " + std::to_string(distinct.size()) +
625 " distinct values, but the maximum is " +
626 std::to_string(value_r_ - value_l_ + 1));
627
628 // Checks if two values in same component are marked as
629 // different.
630 std::set<int> comp_ids;
631 for (int idx : distinct) {
632 if (comp_ids.count(comp_id[idx]))
633 contradiction_error_internal(
634 "sequence", "tried to set two indices as equal and "
635 "different");
636 comp_ids.insert(comp_id[idx]);
637
638 distinct_containing_comp_idx[comp_id[idx]].insert(dist_id);
639 }
640 ++dist_id;
641 }
642 }
643
644 // If some value is in >= 3 sets, then there is a cycle.
645 for (auto &distinct_containing : distinct_containing_comp_idx)
646 if (distinct_containing.size() >= 3)
647 throw error_internal(
648 "failed to generate sequence: complex constraints");
649
650 std::vector<bool> vis_distinct(distinct_constraints_.size(), false);
651 std::vector<bool> initially_defined_comp_idx(comp_count, false);
652
653 // Fills the value in a tree defined by distinct constraints.
654 auto define_tree = [&](int distinct_id) {
655 // The set `distinct_constraints_[distinct_id]` can have some values
656 // that are defined.
657
658 // Generates set of already defined values.
659 std::set<T> defined_values;
660 for (int idx : distinct_constraints_[distinct_id])
661 if (defined_idx[idx]) {
662 // Checks if two values in `distinct_constraints_[dist_id]`
663 // have been set to the same value
664 if (defined_values.count(vec[idx]))
665 contradiction_error_internal(
666 "sequence",
667 "tried to set two indices as equal and different");
668
669 defined_values.insert(vec[idx]);
670 }
671
672 // Generates values in this root distinct constraint.
673 {
674 int new_value_count =
675 distinct_constraints_[distinct_id].size() -
676 static_cast<int>(defined_values.size());
677 std::vector<T> generated_values =
678 generate_distinct_values(new_value_count, defined_values);
679 auto val_it = generated_values.begin();
680 for (int idx : distinct_constraints_[distinct_id])
681 if (defined_idx[idx]) {
682 // The root can cover these components, but there should
683 // not be any other defined in this tree.
684 initially_defined_comp_idx[comp_id[idx]] = false;
685 } else {
686 define_comp(comp_id[idx], *val_it);
687 ++val_it;
688 }
689 }
690
691 // BFS on the tree of distinct constraints.
692 std::queue<std::pair<int, int>> q; // {id, parent id}
693 q.emplace(distinct_id, -1);
694 vis_distinct[distinct_id] = true;
695 while (!q.empty()) {
696 auto [cur_distinct, parent] = q.front();
697 q.pop();
698
699 std::set<int> neigh_distinct;
700 for (int idx : distinct_constraints_[cur_distinct])
701 for (int nxt_distinct :
702 distinct_containing_comp_idx[comp_id[idx]]) {
703 if (nxt_distinct == cur_distinct or
704 nxt_distinct == parent)
705 continue;
706
707 // Cycle found.
708 if (vis_distinct[nxt_distinct])
709 throw error_internal("failed to generate sequence: "
710 "complex constraints");
711
712 neigh_distinct.insert(nxt_distinct);
713 }
714
715 for (int nxt_distinct : neigh_distinct) {
716 vis_distinct[nxt_distinct] = true;
717 q.emplace(nxt_distinct, cur_distinct);
718
719 // Generates this distinct constraint.
720 std::set<T> nxt_defined_values;
721 for (int idx2 : distinct_constraints_[nxt_distinct])
722 if (defined_idx[idx2]) {
723 // There can not be any more defined. This case is
724 // when there are values not coverered by a single
725 // distinct constraint in the tree.
726 if (initially_defined_comp_idx[comp_id[idx2]])
727 throw error_internal(
728 "failed to generate sequence: "
729 "complex constraints");
730
731 nxt_defined_values.insert(vec[idx2]);
732 }
733 int new_value_count =
734 distinct_constraints_[nxt_distinct].size() -
735 static_cast<int>(nxt_defined_values.size());
736 std::vector<T> generated_values = generate_distinct_values(
737 new_value_count, nxt_defined_values);
738 auto val_it = generated_values.begin();
739 for (int idx2 : distinct_constraints_[nxt_distinct])
740 if (!defined_idx[idx2]) {
741 define_comp(comp_id[idx2], *val_it);
742 ++val_it;
743 }
744 }
745 }
746 };
747
748 // Loops through distinct constraints, sorts distinct constraints
749 // by number of defined components (non-increasing). This guarantees
750 // that if there is a valid root (that covers all 'defined'), we find
751 // it.
752 {
753 std::vector<std::pair<int, int>> defined_cnt_and_distinct_idx;
754 int dist_id = 0;
755 for (const std::set<int> &distinct : distinct_constraints_) {
756 int defined_cnt = 0;
757 for (int idx : distinct)
758 if (defined_idx[idx]) {
759 ++defined_cnt;
760 initially_defined_comp_idx[comp_id[idx]] = true;
761 }
762 defined_cnt_and_distinct_idx.emplace_back(defined_cnt, dist_id);
763 ++dist_id;
764 }
765
766 std::sort(defined_cnt_and_distinct_idx.rbegin(),
767 defined_cnt_and_distinct_idx.rend());
768 for (auto [defined_cnt, distinct_idx] :
769 defined_cnt_and_distinct_idx)
770 if (!vis_distinct[distinct_idx])
771 define_tree(distinct_idx);
772 }
773
774 // Loops through distinct constraints do define the rest.
775 for (std::size_t dist_id = 0; dist_id < distinct_constraints_.size();
776 ++dist_id)
777 if (!vis_distinct[dist_id])
778 define_tree(dist_id);
779
780 // Define final values. These values all should be random in [l, r], and
781 // the distinct constraints have already been processed. However, there
782 // can be still equality constraints, so we set entire components.
783 for (int idx = 0; idx < size_; ++idx)
784 if (!defined_idx[idx])
785 define_comp(comp_id[idx], next<T>(value_l_, value_r_));
786
787 if (!values_.empty()) {
788 // Needs to fetch the values from the value set.
789 std::vector<T> value_vec(values_.begin(), values_.end());
790 for (T &val : vec)
791 val = value_vec[val];
792 }
793
794 return instance(vec);
795 }
796};

816// Chooses k values from the sequence, as in a subsequence of size k.
817template <typename INST> INST choose(int k, const INST &inst) {
818 tgen_ensure(0 < k and k <= inst.vec_.size(),
819 "number of elements to choose must be valid");
820 std::vector<typename INST::value_type> new_vec;
821 int need = k;
822 for (int i = 0; need > 0; ++i) {
823 int left = inst.vec_.size() - i;
824 if (next(1, left) <= need) {
825 new_vec.push_back(inst.vec_[i]);
826 need--;
827 }
828 }
829 return INST(new_vec);
830}

840/*
841 * Permutation generation.
842 *
843 * Permutation are defined always as numbers in [0, n), that is, 0-based.
844 */
845
846struct permutation : gen_base<permutation> {
847 int size_; // Size of permutation.
848 std::vector<std::pair<int, int>> sets; // {idx, value}.
849
850 // Creates generator for permutation of size 'size'.
851 permutation(int size) : size_(size) {
852 tgen_ensure(size_ > 0, "size must be positive");
853 }
854
855 // Restricts sequences for permutation[idx] = value.
856 permutation &set(int idx, int value) {
857 tgen_ensure(0 <= idx and idx < size_, "index must be valid");
858 sets.emplace_back(idx, value);
859 return *this;
860 }
861
862 // Permutation instance.
863 // Operations on an instance are not random.
864 struct instance {
865 std::vector<int> vec_; // Permutation.
866 bool add_1_; // If should add 1, for printing.
867
868 instance(const std::vector<int> &vec) : vec_(vec), add_1_(false) {
869 tgen_ensure(!vec_.empty(), "permutation cannot be empty");
870 std::vector<bool> vis(vec_.size(), false);
871 for (std::size_t i = 0; i < vec_.size(); i++) {
872 tgen_ensure(0 <= vec_[i] and
873 vec_[i] < static_cast<int>(vec_.size()),
874 "permutation values must be from `0` to `size-1`");
875 tgen_ensure(!vis[vec_[i]],
876 "cannot have repeated values in permutation");
877 vis[vec_[i]] = true;
878 }
879 }
880 instance(const std::initializer_list<int> &il)
881 : instance(std::vector<int>(il.begin(), il.end())) {}
882
883 // Fetches size.
884 std::size_t size() const { return vec_.size(); }
885
886 // Fetches position idx.
887 int &operator[](int idx) { return vec_[idx]; }
888 const int &operator[](int idx) const { return vec_[idx]; }
889
890 // Returns parity of the permutation (+1 if even, -1 if odd).
891 int parity() const {
892 std::vector<bool> vis(size(), false);
893 int cycles = 0;
894
895 for (std::size_t i = 0; i < size(); ++i)
896 if (!vis[i]) {
897 cycles++;
898 for (int j = i; !vis[j]; j = vec_[j])
899 vis[j] = true;
900 }
901 // Even iff (n - cycles) is even.
902 return ((size() - cycles) % 2 == 0) ? +1 : -1;
903 }
904
905 // Sorts values in increasign order.
906 instance &sort() {
907 std::sort(vec_.begin(), vec_.end());
908 return *this;
909 }
910
911 // Reverses permutation.
912 instance &reverse() {
913 std::reverse(vec_.begin(), vec_.end());
914 return *this;
915 }
916
917 // Inverse of the permutation.
918 instance &inverse() {
919 std::vector<int> inv(size());
920 for (std::size_t i = 0; i < size(); ++i)
921 inv[vec_[i]] = i;
922 swap(vec_, inv);
923 return *this;
924 }
925
926 // Sets that should print values 1-based.
927 instance &add_1() {
928 add_1_ = true;
929 return *this;
930 }
931
932 // Prints in stdout, separated by spaces.
933 friend std::ostream &operator<<(std::ostream &out,
934 const instance &inst) {
935 for (std::size_t i = 0; i < inst.size(); ++i) {
936 if (i > 0)
937 out << ' ';
938 out << inst[i] + inst.add_1_;
939 }
940 return out;
941 }
942
943 // Gets a std::vector representing the instance.
944 std::vector<int> to_std() const { return vec_; }
945 };
946
947 // Generates permutation instance.
948 instance gen() {
949 sequence<int> seq(size_, 0, size_ - 1);
950 seq.distinct();
951 for (auto [idx, val] : sets)
952 seq.set(idx, val);
953 return instance(seq.gen().to_std());
954 }
955
956 // Generates permutation instance, given cycle sizes.
957 instance gen(std::vector<int> cycle_sizes) {
958 tgen_ensure(
959 size_ == std::accumulate(cycle_sizes.begin(), cycle_sizes.end(), 0),
960 "cycle sizes must add up to size of permutation");
961
962 // Creates cycles.
963 std::vector<int> order(size_);
964 std::iota(order.begin(), order.end(), 0);
965 shuffle(order.begin(), order.end());
966 int idx = 0;
967 std::vector<std::vector<int>> cycles;
968 for (int cycle_size : cycle_sizes) {
969 cycles.emplace_back();
970 for (int i = 0; i < cycle_size; ++i)
971 cycles.back().push_back(order[idx++]);
972 }
973
974 // Retrieves permutation from cycles.
975 std::vector<int> perm(size_, -1);
976 for (const std::vector<int> &cycle : cycles) {
977 int cur_size = cycle.size();
978 for (int i = 0; i < cur_size; ++i)
979 perm[cycle[i]] = cycle[(i + 1) % cur_size];
980 }
981
982 return instance(perm);
983 }
