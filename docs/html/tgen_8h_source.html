57// Returns a random number in [l, r].
58template <typename T> T next(T l, T r) {
59 tgen_ensure(l <= r, "range for `next` bust be valid");
60 if constexpr (std::is_integral_v<T>)
61 return std::uniform_int_distribution<T>(l, r)(rng_internal);
62 else if constexpr (std::is_floating_point_v<T>)
63 return std::uniform_real_distribution<T>(l, r)(rng_internal);
64 else
65 throw error_internal("invalid type for next (" +
66 std::string(typeid(T).name()) + ")");
67}

69// Shuffles [first, last) inplace uniformly.
70template <typename It> void shuffle(It first, It last) {
71 if (first == last)
72 return;
73
74 for (It i = first + 1; i != last; ++i)
75 std::iter_swap(i, first + next(0, int(i - first)));
76}
77
78// Shuffles container uniformly. Returns a copy.
79template <typename C> C shuffle(C container) {
80 shuffle(container.begin(), container.end());
81 return container;
82}
83
84// Returns a random element from [first, last).
85template <typename It> typename It::value_type any(It first, It last) {
86 int size = std::distance(first, last);
87 It it = first;
88 std::advance(it, next(0, size - 1));
89 return *it;
90}

92// Returns a random element from container.
93template <typename C> typename C::value_type any(const C &container) {
94 return any(container.begin(), container.end());

97 return any(std::vector<T>(list.begin(), list.end()));

100// Chooses k values from the container, as in a subsequence of size k. Returns a
101// copy.
102template <typename C> C choose(int k, const C &container) {
103 tgen_ensure(0 < k and k <= container.size(),
104 "number of elements to choose must be valid");
105 std::vector<typename C::value_type> new_vec;
106 C new_container;
107 int need = k, left = container.size();
108 for (auto cur_it = container.begin(); cur_it != container.end(); ++cur_it) {
109 if (next(1, left--) <= need) {
110 new_container.insert(new_container.end(), *cur_it);
111 need--;
112 }
113 }
114 return new_container;
115}
116template <typename T>
117std::vector<T> choose(int k, const std::initializer_list<T> &list) {
118 return choose(k, std::vector<T>(list.begin(), list.end()));
119}

127/*
128 * Opts - options given to the generator.
129 *
130 * Incompatible with testlib.
131 *
132 * Opts are a list of either positional or named options.
133 *
134 * Named options is given in one of the following formats:
135 * 1) -keyname=value or --keyname=value (ex. -n=10 , --test-count=20)
136 * 2) -keyname value or --keyname value (ex. -n 10 , --test-count 20)
137 *
138 * Positional options are number from 0 sequentially.
139 * For example, for "10 -n=20 str" positional option 1 is the string "str".
140 */
141
142inline std::vector<std::string>
143 pos_opts_internal; // Dictionary containing the positional parsed opts.
144inline std::map<std::string, std::string>
145 named_opts_internal; // Global dictionary the named parsed opts.
146
147// Returns true if there is an opt at a given index.
148inline bool has_opt(int index) {
149 return 0 <= index and index < pos_opts_internal.size();
150}
151
152// Returns true if there is an opt with a given key.
153inline bool has_opt(const std::string &key) {
154 return named_opts_internal.count(key) != 0;
155}
156
157template <typename T> T get_opt_internal(const std::string &value) {
158 try {
159 if constexpr (std::is_same_v<T, bool>) {
160 if (value == "true" or value == "1")
161 return true;
162 if (value == "false" or value == "0")
163 return false;
164 } else if constexpr (std::is_integral_v<T>) {
165 if constexpr (std::is_unsigned_v<T>)
166 return static_cast<T>(std::stoull(value));
167 else
168 return static_cast<T>(std::stoll(value));
169 } else if constexpr (std::is_floating_point_v<T>)
170 return static_cast<T>(std::stold(value));
171 else
172 return value; // default: std::string
173 } catch (...) {
174 }
175
176 throw error_internal("invalid value `" + value + "` for type " +
177 typeid(T).name());
178}
179
180// Returns the parsed opt by a given key. If no opts with the given key are
181// found, returns the given default_value.
182template <typename T, typename KEY>
183T opt(const KEY &key, std::optional<T> default_value = std::nullopt) {
184 if constexpr (std::is_same_v<KEY, int>) {
185 if (!has_opt(key)) {
186 if (default_value)
187 return *default_value;
188 throw error_internal("cannot find key with index " +
189 std::to_string(key));
190 }
191 return get_opt_internal<T>(pos_opts_internal[key]);
192 } else { // std::string
193 if (!has_opt(key)) {
194 if (default_value)
195 return *default_value;
196 throw error_internal("cannot find key with key " +
197 std::string(key));
198 }
199 return get_opt_internal<T>(named_opts_internal[key]);
200 }
201}
202
203inline void parse_opts_internal(int argc, char **argv) {
204 // Parses the opts into `pos_opts_internal` vector and `named_opts_internal`
205 // map. Starting from 1 to ignore the name of the executable.
206 for (int i = 1; i < argc; i++) {
207 std::string key(argv[i]);
208
209 if (key[0] == '-') {
210 tgen_ensure(key.size() > 1,
211 "invalid opt (" + std::string(argv[i]) + ")");
212 if ('0' <= key[1] and key[1] <= '9') {
213 // This case is a positional negative number argument
214 pos_opts_internal.push_back(key);
215 continue;
216 }
217
218 // pops first char '-'
219 key = key.substr(1);
220 } else {
221 // This case is a positional argument that does not start with '-'
222 pos_opts_internal.push_back(key);
223 continue;
224 }
225
226 // Pops a possible second char '-'.
227 if (key[0] == '-') {
228 tgen_ensure(key.size() > 1,
229 "invalid opt (" + std::string(argv[i]) + ")");
230
231 // pops first char '-'
232 key = key.substr(1);
233 }
234
235 // Assumes that, if it starts with '-' and second char is not a digit,
236 // then it is a <key, value> pair.
237 // 1 or 2 chars '-' have already been poped.
238
239 std::size_t eq = key.find('=');
240 if (eq != std::string::npos) {
241 // This is the '--key=value' case.
242 std::string value = key.substr(eq + 1);
243 key = key.substr(0, eq);
244 tgen_ensure(!key.empty() and !value.empty(),
245 "expected non-empty key/value in opt (" +
246 std::string(argv[1]));
247 tgen_ensure(named_opts_internal.count(key) == 0,
248 "cannot have repeated keys");
249 named_opts_internal[key] = value;
250 } else {
251 // This is the '--key value' case.
252 tgen_ensure(named_opts_internal.count(key) == 0,
253 "cannot have repeated keys");
254 tgen_ensure(argv[i + 1], "value cannot be empty");
255 named_opts_internal[key] = std::string(argv[i + 1]);
256 i++;
257 }
258 }
259}
260inline void set_seed_internal(int argc, char **argv) {
261 std::vector<uint32_t> seed;
262
263 // Starting from 1 to ignore the name of the executable.
264 for (int i = 1; i < argc; ++i) {
265 // We append the number of chars, and then the list of chars.
266 int size_pos = seed.size();
267 seed.push_back(0);
268 for (char *s = argv[i]; *s != '\0'; ++s) {
269 ++seed[size_pos];
270 seed.push_back(*s);
271 }
272 }
273 std::seed_seq seq(seed.begin(), seed.end());
274 rng_internal.seed(seq);
275}
276
277// Registers generator by initializing rnd and parsing opts.
278inline void register_gen(int argc, char **argv) {
279 set_seed_internal(argc, argv);
280
281 pos_opts_internal.clear();
282 named_opts_internal.clear();
283 parse_opts_internal(argc, argv);
284}

296template <typename T> struct sequence {
297 int size_; // Size of sequence.
298 T value_l_, value_r_; // Range of defined values.
299 std::set<T> values_; // Set of values. If empty, use range. if not,
300 // represents the possible values, and the range
301 // represents the index in this set)
302 std::map<T, int>
303 value_idx_in_set_; // Index of every value in the set above.
304 std::vector<std::pair<T, T>> val_range_; // Range of values of each index.
305 std::vector<std::vector<int>> neigh_; // Adjacency list of equality.
306 std::vector<std::set<int>>
307 distinct_constraints_; // All distinct constraints.
308
309 // Creates generator for sequences of size 'size', with random T in [l, r]
310 sequence(int size, T value_l, T value_r)
311 : size_(size), value_l_(value_l), value_r_(value_r), neigh_(size) {
312 tgen_ensure(size_ > 0, "size must be positive");
313 tgen_ensure(value_l_ <= value_r_, "value range must be valid");
314 for (int i = 0; i < size_; ++i)
315 val_range_.emplace_back(value_l_, value_r_);
316 }
317
318 // Creates sequence with value set.
319 sequence(int size, const std::set<T> &values)
320 : size_(size), values_(values), neigh_(size) {
321 tgen_ensure(size_ > 0, "size must be positive");
322 tgen_ensure(!values.empty(), "value set must be non-empty");
323 value_l_ = 0, value_r_ = values.size() - 1;
324 for (int i = 0; i < size_; ++i)
325 val_range_.emplace_back(value_l_, value_r_);
326 int idx = 0;
327 for (T value : values_)
328 value_idx_in_set_[value] = idx++;
329 }
330 sequence(int size, const std::vector<T> &values)
331 : sequence(size, std::set<T>(values.begin(), values.end())) {}
332 sequence(int size, const std::initializer_list<T> &values)
333 : sequence(size, std::set<T>(values.begin(), values.end())) {}
334
335 // Restricts sequences for sequence[idx] = value.
336 sequence &set(int idx, T value) {
337 tgen_ensure(0 <= idx and idx < size_, "index must be valid");
338 if (values_.size() == 0) {
339 auto &[left, right] = val_range_[idx];
340 tgen_ensure(left <= value and value <= right,
341 "value must be in the defined range");
342 left = right = value;
343 } else {
344 tgen_ensure(values_.count(value),
345 "value must be in the set of values");
346 auto &[left, right] = val_range_[idx];
347 int new_val = value_idx_in_set_[value];
348 tgen_ensure(left <= new_val and new_val <= right,
349 "must not set to two different values");
350 left = right = new_val;
351 }
352 return *this;
353 }
354
355 // Restricts sequences for sequence[idx_1] = sequence[idx_2].
356 sequence &equal(int idx_1, int idx_2) {
357 tgen_ensure(0 <= std::min(idx_1, idx_2) and
358 std::max(idx_1, idx_2) < size_,
359 "indices must be valid");
360 if (idx_1 == idx_2)
361 return *this;
362
363 neigh_[idx_1].push_back(idx_2);
364 neigh_[idx_2].push_back(idx_1);
365 return *this;
366 }
367
368 // Restricts sequences for sequence[left..right] to have all equal values.
369 sequence &equal_range(int left, int right) {
370 tgen_ensure(0 <= left and left <= right and right < size_,
371 "range indices bust be valid");
372 for (int i = left; i < right; ++i)
373 equal(i, i + 1);
374 return *this;
375 }
376
377 // Restricts sequences for sequence[S] to be distinct, for given subset S of
378 // indices.
379 // You can not add two of these restrictions with intersection.
380 sequence &distinct(const std::set<int> &indices) {
381 distinct_constraints_.push_back(indices);
382 return *this;
383 }
384 sequence &distinct(const std::initializer_list<int> &indices) {
385 return distinct(std::set<int>(indices.begin(), indices.end()));
386 }
387
388 // Restricts sequences for sequence[idx_1] != sequence[idx_2].
389 sequence &different(int idx_1, int idx_2) {
390 std::set<int> indices = {idx_1, idx_2};
391 return distinct(indices);
392 }
393
394 // Restricts sequences with distinct elements.
395 sequence &distinct() {
396 std::set<int> indices;
397 for (int i = 0; i < size_; ++i)
398 indices.insert(i);
399 return distinct(indices);
400 }

454 // Generates a uniformly random list of k distinct values in `[value_l,
455 // value_r]`, such that no value is in `forbidden_values`.
456 std::vector<T>
457 generate_distinct_values(int k, const std::set<T> &forbidden_values) {
458 for (auto forbidden : forbidden_values)
459 tgen_ensure(value_l_ <= forbidden and forbidden <= value_r_);
460 // We generate our numbers in the range [0, num_available) with
461 // num_available = (r-l+1)-(forbidden_values.size()), and then map them
462 // to the correct range. We will run k steps of Fisherâ€“Yates, using a
463 // map to store a virtual sequence that starts with a[i] = i.
464 T num_available = (value_r_ - value_l_ + 1) - forbidden_values.size();
465 if (num_available < k)
466 throw error_internal(
467 "failed to generate sequence: complex constraints");
468 std::map<T, T> virtual_list;
469 std::vector<T> list;
470 for (int i = 0; i < k; i++) {
471 T j = next<T>(i, num_available - 1);
472 T vj = virtual_list.count(j) ? virtual_list[j] : j;
473 T vi = virtual_list.count(i) ? virtual_list[i] : i;
474
475 virtual_list[j] = vi, virtual_list[i] = vj;
476
477 list.push_back(virtual_list[i]);
478 }
479
480 // Shifts back to correct range, but there might still be values
481 // that we can not use.
482 for (T &value : list)
483 value += value_l_;
484
485 // Now for every generated value, we shift it by how many forbidden
486 // values are <= to it.
487 std::vector<std::pair<T, int>> values_sorted;
488 for (int i = 0; i < list.size(); ++i)
489 values_sorted.emplace_back(list[i], i);
490 // We iterate through them in increasing order.
491 std::sort(values_sorted.begin(), values_sorted.end());
492 auto cur_it = forbidden_values.begin();
493 int smaller_forbidden_count = 0;
494 for (auto [val, idx] : values_sorted) {
495 while (cur_it != forbidden_values.end() and
496 *cur_it <= val + smaller_forbidden_count)
497 ++cur_it, ++smaller_forbidden_count;
498 list[idx] += smaller_forbidden_count;
499 }
500
501 return list;
502 }

504 // Generate sequence instance.
505 instance gen() {
506 std::vector<T> vec(size_);
507 std::vector<bool> defined_idx(
508 size_, false); // For every index, if it has been set in `vec`.
509
510 std::vector<int> comp_id(size_, -1); // Component id of each index.
511 std::vector<std::vector<int>> comp(size_); // Component of each comp-id.
512 int comp_count = 0; // Number of different components.
513
514 // Defines value of entire component.
515 auto define_comp = [&](int cur_comp, T val) {
516 for (int idx : comp[cur_comp]) {
517 tgen_ensure(!defined_idx[idx]);
518 vec[idx] = val;
519 defined_idx[idx] = true;
520 }
521 };
522
523 // Groups = components.
524 {
525 std::vector<bool> vis(size_, false); // Visited for each index.
526 for (int idx = 0; idx < size_; ++idx)
527 if (!vis[idx]) {
528 T new_value;
529 bool value_defined = false;
530
531 // BFS to visit the connected component, grouping equal
532 // values.
533 std::queue<int> q({idx});
534 vis[idx] = true;
535 std::vector<int> component;
536 while (q.size()) {
537 int cur_idx = q.front();
538 q.pop();
539
540 component.push_back(cur_idx);
541
542 // Checks value.
543 auto [l, r] = val_range_[cur_idx];
544 if (l == r) {
545 if (!value_defined) {
546 // We found the value.
547 value_defined = true;
548 new_value = l;
549 } else if (new_value != l) {
550 // We found a contradiction
551 contradiction_error_internal(
552 "sequence",
553 "tried to set value to `" +
554 std::to_string(new_value) +
555 "`, but it was already set as `" +
556 std::to_string(l) + "`");
557 }
558 }
559
560 for (int nxt_idx : neigh_[cur_idx]) {
561 if (!vis[nxt_idx]) {
562 vis[nxt_idx] = true;
563 q.push(nxt_idx);
564 }
565 }
566 }
567
568 // Group entire component, checking if value is defined.
569 for (int cur_idx : component) {
570 comp_id[cur_idx] = comp_count;
571 comp[comp_id[cur_idx]].push_back(cur_idx);
572 }
573
574 // Sets value if needed.
575 if (value_defined)
576 define_comp(comp_count, new_value);
577
578 ++comp_count;
579 }
580 }
581
582 // Initial parsing of distinct constraints.
583 std::vector<std::set<int>> distinct_containing_comp_idx(comp_count);
584 {
585 int dist_id = 0;
586 for (const std::set<int> &distinct : distinct_constraints_) {
587 // Checks if there are too many distinct values.
588 if (distinct.size() > value_r_ - value_l_ + 1)
589 contradiction_error_internal(
590 "sequence",
591 "tried to generate " + std::to_string(distinct.size()) +
592 " distinct values, but the maximum is " +
593 std::to_string(value_r_ - value_l_ + 1));
594
595 // Checks if two values in same component are marked as
596 // different.
597 std::set<int> comp_ids;
598 for (int idx : distinct) {
599 if (comp_ids.count(comp_id[idx]))
600 contradiction_error_internal(
601 "sequence", "tried to set two indices as equal and "
602 "different");
603 comp_ids.insert(comp_id[idx]);
604
605 distinct_containing_comp_idx[comp_id[idx]].insert(dist_id);
606 }
607 ++dist_id;
608 }
609 }
610
611 // If some value is in >= 3 sets, then there is a cycle.
612 for (auto &distinct_containing : distinct_containing_comp_idx)
613 if (distinct_containing.size() >= 3)
614 throw error_internal(
615 "failed to generate sequence: complex constraints");
616
617 std::vector<bool> vis_distinct(distinct_constraints_.size(), false);
618 std::vector<bool> initially_defined_comp_idx(comp_count, false);
619
620 // Fills the value in a tree defined by distinct constraints.
621 auto define_tree = [&](int distinct_id) {
622 // The set `distinct_constraints_[distinct_id]` can have some values
623 // that are defined.
624
625 // Generates set of already defined values.
626 std::set<T> defined_values;
627 for (int idx : distinct_constraints_[distinct_id])
628 if (defined_idx[idx]) {
629 // Checks if two values in `distinct_constraints_[dist_id]`
630 // have been set to the same value
631 if (defined_values.count(vec[idx]))
632 contradiction_error_internal(
633 "sequence",
634 "tried to set two indices as equal and different");
635
636 defined_values.insert(vec[idx]);
637 }
638
639 // Generates values in this root distinct constraint.
640 {
641 int new_value_count =
642 distinct_constraints_[distinct_id].size() -
643 int(defined_values.size());
644 std::vector<T> generated_values =
645 generate_distinct_values(new_value_count, defined_values);
646 auto val_it = generated_values.begin();
647 for (int idx : distinct_constraints_[distinct_id])
648 if (defined_idx[idx]) {
649 // The root can cover these components, but there should
650 // not be any other defined in this tree.
651 initially_defined_comp_idx[comp_id[idx]] = false;
652 } else {
653 define_comp(comp_id[idx], *val_it);
654 ++val_it;
655 }
656 }
657
658 // BFS on the tree of distinct constraints.
659 std::queue<std::pair<int, int>> q; // {id, parent id}
660 q.emplace(distinct_id, -1);
661 vis_distinct[distinct_id] = true;
662 while (q.size()) {
663 auto [cur_distinct, parent] = q.front();
664 q.pop();
665
666 std::set<int> neigh_distinct;
667 for (int idx : distinct_constraints_[cur_distinct])
668 for (int nxt_distinct :
669 distinct_containing_comp_idx[comp_id[idx]]) {
670 if (nxt_distinct == cur_distinct or
671 nxt_distinct == parent)
672 continue;
673
674 // Cycle found.
675 if (vis_distinct[nxt_distinct])
676 throw error_internal("failed to generate sequence: "
677 "complex constraints");
678
679 neigh_distinct.insert(nxt_distinct);
680 }
681
682 for (int nxt_distinct : neigh_distinct) {
683 vis_distinct[nxt_distinct] = true;
684 q.emplace(nxt_distinct, cur_distinct);
685
686 // Generates this distinct constraint.
687 std::set<T> nxt_defined_values;
688 for (int idx2 : distinct_constraints_[nxt_distinct])
689 if (defined_idx[idx2]) {
690 // There can not be any more defined. This case is
691 // when there are values not coverered by a single
692 // distinct constraint in the tree.
693 if (initially_defined_comp_idx[comp_id[idx2]])
694 throw error_internal(
695 "failed to generate sequence: "
696 "complex constraints");
697
698 nxt_defined_values.insert(vec[idx2]);
699 }
700 int new_value_count =
701 distinct_constraints_[nxt_distinct].size() -
702 int(nxt_defined_values.size());
703 std::vector<T> generated_values = generate_distinct_values(
704 new_value_count, nxt_defined_values);
705 auto val_it = generated_values.begin();
706 for (int idx2 : distinct_constraints_[nxt_distinct])
707 if (!defined_idx[idx2]) {
708 define_comp(comp_id[idx2], *val_it);
709 ++val_it;
710 }
711 }
712 }
713 };
714
715 // Loops through distinct constraints, sorts distinct constraints
716 // by number of defined components (non-increasing). This guarantees
717 // that if there is a valid root (that covers all 'defined'), we find
718 // it.
719 {
720 std::vector<std::pair<int, int>> defined_cnt_and_distinct_idx;
721 int dist_id = 0;
722 for (const std::set<int> &distinct : distinct_constraints_) {
723 int defined_cnt = 0;
724 for (int idx : distinct)
725 if (defined_idx[idx]) {
726 ++defined_cnt;
727 initially_defined_comp_idx[comp_id[idx]] = true;
728 }
729 defined_cnt_and_distinct_idx.emplace_back(defined_cnt, dist_id);
730 ++dist_id;
731 }
732
733 std::sort(defined_cnt_and_distinct_idx.rbegin(),
734 defined_cnt_and_distinct_idx.rend());
735 for (auto [defined_cnt, distinct_idx] :
736 defined_cnt_and_distinct_idx)
737 if (!vis_distinct[distinct_idx])
738 define_tree(distinct_idx);
739 }
740
741 // Loops through distinct constraints do define the rest.
742 for (int dist_id = 0; dist_id < distinct_constraints_.size(); ++dist_id)
743 if (!vis_distinct[dist_id])
744 define_tree(dist_id);
745
746 // Define final values. These values all should be random in [l, r], and
747 // the distinct constraints have already been processed. However, there
748 // can be still equality constraints, so we set entire components.
749 for (int idx = 0; idx < size_; ++idx)
750 if (!defined_idx[idx])
751 define_comp(comp_id[idx], next<T>(value_l_, value_r_));
752
753 if (!values_.empty()) {
754 // Needs to fetch the values from the value set.
755 std::vector<T> value_vec(values_.begin(), values_.end());
756 for (T &val : vec)
757 val = value_vec[val];
758 }
759
760 return instance(vec);
761 }
762
763 // Calls the generator until predicate is true.
764 template <typename PRED>
765 instance gen_until(PRED predicate, int max_tries,
766 bool random_default = false) {
767 for (int i = 0; i < max_tries; ++i) {
768 instance inst = gen();
769 if (predicate(inst))
770 return inst;
771 }
772 if (random_default)
773 return gen();
774 else
775 throw error_internal(
776 "could not generate instance matching predicate");
777 }
778};

798// Chooses k values from the sequence, as in a subsequence of size k.
799template <typename INST> INST choose(int k, const INST &inst) {
800 tgen_ensure(0 < k and k <= inst.vec_.size(),
801 "number of elements to choose must be valid");
802 std::vector<typename INST::value_type> new_vec;
803 int need = k;
804 for (int i = 0; need > 0; ++i) {
805 int left = inst.vec_.size() - i;
806 if (next(1, left) <= need) {
807 new_vec.push_back(inst.vec_[i]);
808 need--;
809 }
810 }
811 return INST(new_vec);
812}
