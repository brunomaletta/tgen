80// Shuffles [first, last) inplace uniformly.
81template <typename It> void shuffle(It first, It last) {
82 if (first == last)
83 return;
84
85 for (It i = first + 1; i != last; ++i)
86 std::iter_swap(i, first + next(0, int(i - first)));
87}

89// Shuffles container uniformly. Returns a copy.
90template <typename C> C shuffle(C container) {
91 shuffle(container.begin(), container.end());
92 return container;
93}

111// Chooses k values from the container, as in a subsequence of size k. Returns a
112// copy.
113template <typename C> C choose(int k, const C &container) {
114 tgen_ensure(0 < k and k <= container.size(),
115 "number of elements to choose must be valid");
116 std::vector<typename C::value_type> new_vec;
117 C new_container;
118 int need = k, left = container.size();
119 for (auto cur_it = container.begin(); cur_it != container.end(); ++cur_it) {
120 if (next(1, left--) <= need) {
121 new_container.insert(new_container.end(), *cur_it);
122 need--;
123 }
124 }
125 return new_container;
126}
127template <typename T>
128std::vector<T> choose(int k, const std::initializer_list<T> &list) {
129 return choose(k, std::vector<T>(list.begin(), list.end()));

138/*
139 * Opts - options given to the generator.
140 *
141 * Incompatible with testlib.
142 *
143 * Opts are a list of either positional or named options.
144 *
145 * Named options is given in one of the following formats:
146 * 1) -keyname=value or --keyname=value (ex. -n=10 , --test-count=20)
147 * 2) -keyname value or --keyname value (ex. -n 10 , --test-count 20)
148 *
149 * Positional options are number from 0 sequentially.
150 * For example, for "10 -n=20 str" positional option 1 is the string "str".
151 */
152
153std::vector<std::string>
154 __pos_opts; // Dictionary containing the positional parsed opts.
155std::map<std::string, std::string>
156 __named_opts; // Global dictionary the named parsed opts.
157
158// Returns true if there is an opt at a given index.
159bool has_opt(int index) { return 0 <= index and index < __pos_opts.size(); }
160
161// Returns true if there is an opt with a given key.
162bool has_opt(const std::string &key) { return __named_opts.count(key) != 0; }
163
164template <typename T>
165T __get_opt(const std::string &value, std::true_type, std::false_type,
166 std::false_type) {
167 // Parses 'value' into bool.
168 if (value == "true" or value == "1")
169 return true;
170 if (value == "false" or value == "0")
171 return false;
172 throw __error("invalid value " + value + " for type bool");
173}
174template <typename T>
175T __get_opt(const std::string &value, std::false_type, std::true_type,
176 std::false_type) {
177 // Parses 'value' into integral type T.
178 if (std::is_unsigned<T>())
179 return static_cast<T>(std::stoull(value));
180 return static_cast<T>(std::stoll(value));
181}
182template <typename T>
183T __get_opt(const std::string &value, std::false_type, std::false_type,
184 std::true_type) {
185 // Parses 'value' into floating type T.
186 return static_cast<T>(std::stold(value));
187}
188template <typename T>
189T __get_opt(const std::string &value, std::false_type, std::false_type,
190 std::false_type) {
191 // Parses 'value' into std::string.
192 return value;
193}
194template <typename T> T __get_opt(const std::string &value) {
195 // Parses 'value' into type T.
196 try {
197 return __get_opt<T>(value, std::is_same<T, bool>(),
198 std::is_integral<T>(), std::is_floating_point<T>());
199 } catch (const std::invalid_argument &er) {
200 }
201 throw __error("invalid value " + value + " for type " + typeid(T).name());
202}
203
204// Returns the parsed opt by a given index.
205template <typename T> T opt(int index) {
206 tgen_ensure(has_opt(index), "cannot find key with index " + index);
207 return __get_opt<T>(__pos_opts[index]);
208}

213 if (!has_opt(index))

232char __fetch_char(char *s, int idx) {
233 tgen_ensure(s[idx] != '\n', "tried to fetch end of string");
234 return s[idx];
235}
236std::string __read_until(char *s) {
237 // Reads non-empty string until it hits a ' ' or the string ends.
238 std::string read_str;
239 int idx = 0;
240 while (s[idx] != '\0') {
241 char nxt_char = __fetch_char(s, idx++);
242 if (nxt_char == ' ')
243 break;
244 read_str += nxt_char;
245 }
246
247 tgen_ensure(read_str.size() > 0, "read string cannot be empty");
248 return read_str;
249}
250void __parse_opts(int argc, char **argv) {
251 // Parses the opts into __pos_opts vector and __named_opts map.
252 // Starting from 1 to ignore the name of the executable.
253 for (int i = 1; i < argc; i++) {
254 std::string key = __read_until(argv[i]);
255
256 if (key[0] == '-') {
257 tgen_ensure(key.size() > 1,
258 "invalid opt (" + std::string(argv[i]) + ")");
259 if (isdigit(key[1])) {
260 // This case is a positional negative number argument
261 __pos_opts.push_back(key);
262 continue;
263 }
264
265 // pops first char '-'
266 key = key.substr(1);
267 } else {
268 // This case is a positional argument that does not start with '-'
269 __pos_opts.push_back(key);
270 continue;
271 }
272
273 // Pops a possible second char '-'.
274 if (key[0] == '-') {
275 tgen_ensure(key.size() > 1,
276 "invalid opt (" + std::string(argv[i]) + ")");
277
278 // pops first char '-'
279 key = key.substr(1);
280 }
281
282 // Assumes that, if it starts with '-' and second char is not a digit,
283 // then it is a <key, value> pair.
284 // 1 or 2 chars '-' have already been poped.
285
286 size_t eq = key.find('=');
287 if (eq != std::string::npos) {
288 // This is the '--key=value' case.
289 std::string value = key.substr(eq + 1);
290 key = key.substr(0, eq);
291 tgen_ensure(
292 key.size() > 0 and value.size() > 0, "expected non-empty\
293 key/value in opt (" + std::string(argv[1]));
294 tgen_ensure(__named_opts.count(key) == 0,
295 "cannot have repeated keys");
296 __named_opts[key] = value;
297 } else {
298 // This is the '--key value' case.
299 tgen_ensure(__named_opts.count(key) == 0,
300 "cannot have repeated keys");
301 tgen_ensure(argv[i + 1], "value cannot be empty");
302 __named_opts[key] = __read_until(argv[i + 1]);
303 i++;
304 }
305 }
306}

308// Registers generator by initializing rnd and parsing opts.
309void register_gen(int argc, char **argv) {
310 std::vector<uint32_t> seed;
311
312 // Starting from 1 to ignore the name of the executable.
313 for (int i = 1; i < argc; ++i) {
314 // We append the number of chars, and then the list of chars.
315 int size_pos = seed.size();
316 seed.push_back(0);
317 for (char *s = argv[i]; *s != '\0'; ++s) {
318 ++seed[size_pos];
319 seed.push_back(*s);
320 }
321 }
322 std::seed_seq seq(seed.begin(), seed.end());
323 __rng.seed(seq);
324
325 __parse_opts(argc, argv);
326}

338template <typename T> struct sequence {
339 int size; // Size of sequence.
340 T value_l, value_r; // Range of defined values.
341 std::set<T> values; // Set of values. If empty, use range. if not,
342 // represents the possible values, and the range
343 // represents the index in this set)
344 std::map<T, int> value_idx_in_set; // Index of every value in the set above.
345 std::vector<std::pair<T, T>> val_range; // Range of values of each index.
346 std::vector<std::vector<int>> neigh; // Adjacency list of equality.
347 std::set<int>
348 idx_distinct_constraints; // Indices in some distinction constraint.
349 std::vector<std::set<int>>
350 distinct_constraints; // All distinct constraints.
351
352 // Creates generator for sequences of size 'size', with random T in [l, r]
353 sequence(int size, T value_l, T value_r)
354 : size(size), value_l(value_l), value_r(value_r), neigh(size) {
355 tgen_ensure(value_l <= value_r, "value range must be valid");
356 for (int i = 0; i < size; ++i)
357 val_range.emplace_back(value_l, value_r);
358 }
359
360 // Creates sequence with value set.
361 sequence(int size, const std::set<T> &values)
362 : size(size), values(values), neigh(size) {
363 tgen_ensure(values.size() > 0, "must have at least one value");
364 value_l = 0, value_r = values.size() - 1;
365 for (int i = 0; i < size; ++i)
366 val_range.emplace_back(value_l, value_r);
367 int idx = 0;
368 for (T value : values)
369 value_idx_in_set[value] = idx++;
370 }
371 sequence(int size, const std::vector<T> &values)
372 : sequence(size, std::set<T>(values.begin(), values.end())) {}
373 sequence(int size, const std::initializer_list<T> &values)
374 : sequence(size, std::set<T>(values.begin(), values.end())) {}
375
376 // Restricts sequences for sequence[idx] = value.
377 sequence &set(int idx, T value) {
378 tgen_ensure(0 <= idx and idx < size, "index must be valid");
379 if (values.size() == 0) {
380 auto &[left, right] = val_range[idx];
381 tgen_ensure(left <= value and value <= right,
382 "value must be in the defined range");
383 left = right = value;
384 } else {
385 tgen_ensure(values.count(value),
386 "value must be in the set of values");
387 auto &[left, right] = val_range[idx];
388 int new_val = value_idx_in_set[value];
389 tgen_ensure(left <= new_val and new_val <= right,
390 "must not set to two different values");
391 left = right = new_val;
392 }
393 return *this;
394 }
395
396 // Restricts sequences for sequence[idx_1] = sequence[idx_2].
397 sequence &equal(int idx_1, int idx_2) {
398 tgen_ensure(0 <= std::min(idx_1, idx_2) and
399 std::max(idx_1, idx_2) < size,
400 "indices must be valid");
401 if (idx_1 == idx_2)
402 return *this;
403
404 neigh[idx_1].push_back(idx_2);
405 neigh[idx_2].push_back(idx_1);
406 return *this;
407 }
408
409 // Restricts sequences for sequence[left..right] to have all equal values.
410 sequence &equal_range(int left, int right) {
411 tgen_ensure(0 <= left and left <= right and right < size,
412 "range indices bust be valid");
413 for (int i = left; i < right; ++i)
414 equal(i, i + 1);
415 return *this;
416 }
417
418 // Restricts sequences for sequence[S] to be distinct, for given subset S of
419 // indices.
420 // You can not add two of these restrictions with intersection.
421 sequence &distinct(const std::set<int> &indices) {
422 for (int idx : indices) {
423 if (idx_distinct_constraints.count(idx))
424 throw __error(
425 "cannot add same index in two distinct constraints");
426 idx_distinct_constraints.insert(idx);
427 }
428
429 distinct_constraints.push_back(indices);
430 return *this;
431 }
432 sequence &distinct(const std::initializer_list<int> &indices) {
433 return distinct(std::set<int>(indices.begin(), indices.end()));
434 }
435
436 // Restricts sequences for sequence[idx_1] != sequence[idx_2]
437 sequence &different(int idx_1, int idx_2) {
438 std::set<int> indices = {idx_1, idx_2};
439 distinct(indices);
440 return *this;
441 }
442
443 // Restricts sequences with distinct elements.
444 sequence &distinct() {
445 std::set<int> indices;
446 for (int i = 0; i < size; ++i)
447 indices.insert(i);
448 distinct(indices);
449 return *this;
450 }

452 // Sequence instance.
453 // Operations on an instance are not random.
454 struct instance {
455 using value_type = T; // Value type, for templates.
456 std::vector<T> vec; // Sequence.
457
458 instance(const std::vector<T> &vec) : vec(vec) {}
459 instance(const std::initializer_list<T> &list)
460 : vec(list.begin(), list.end()) {}
461
462 // Fetches size.
463 size_t size() { return vec.size(); }
464
465 // Fetches position idx.
466 T operator[](int idx) const { return vec[idx]; }
467
468 // Sorts values in increasign order.
469 instance &sort() {
470 std::sort(vec.begin(), vec.end());
471 return *this;
472 }
473
474 // Reverses sequence.
475 instance &reverse() {
476 std::reverse(vec.begin(), vec.end());
477 return *this;
478 }
479

485 out << ' ';

489 }

494
495 // Generate sequence instance.
496 instance gen() {
497 std::vector<T> vec(size);
498
499 std::vector<int> comp_id(size, -1); // component id of each index.
500 std::vector<std::vector<int>> comp(size); // component of each comp-id.
501 int comp_count = 0; // number of different components.
502 std::vector<bool> vis(size, false);
503 for (int i = 0; i < size; ++i)
504 if (!vis[i]) {
505 T new_value;
506 bool value_set = false;
507
508 // BFS to visit the connected component, grouping equal values.
509 std::queue<int> q;
510 q.push(i);
511 vis[i] = true;
512 std::vector<int> component;
513 while (q.size()) {
514 int j = q.front();
515 q.pop();
516
517 component.push_back(j);
518
519 // Checks value.
520 auto [l, r] = val_range[j];
521 if (l == r) {
522 if (!value_set) {
523 // We found the value.
524 value_set = true;
525 new_value = l;
526 } else if (new_value != l) {
527 // We found a contradiction
528 std::stringstream ss;
529 ss << "tried to set value to `" << new_value
530 << "`, but it was already set as `" << l << "`";
531 __contradiction_error("sequence", ss.str());
532 }
533 }
534
535 for (int k : neigh[j]) {
536 if (!vis[k]) {
537 vis[k] = true;
538 q.push(k);
539 }
540 }
541 }
542
543 // Group entire component, checking if value is defined.
544 for (int j : component) {
545 if (value_set)
546 val_range[j].first = val_range[j].second = new_value;
547 comp_id[j] = comp_count;
548 comp[comp_id[j]].push_back(j);
549 }
550 ++comp_count;
551 }
552
553 // Adds artificial distinct constraints.
554 for (int i = 0; i < size; ++i)
555 if (!idx_distinct_constraints.count(i)) {
556 std::set<int> cur = {i};
557 distinct_constraints.push_back(cur);
558 }
559
560 // Loops through distinct constraints.
561 for (const std::set<int> &distinct : distinct_constraints) {
562 // Checks if there are too many distinct values.
563 if (distinct.size() > value_r - value_l + 1)
564 __contradiction_error(
565 "sequence", "tried to generate " +
566 std::to_string(distinct.size()) +
567 " distinct values, but the maximum is " +
568 std::to_string(value_r - value_l + 1));
569
570 // Checks if two values in same component are marked as different.
571 std::set<int> comp_ids;
572 for (int idx : distinct) {
573 if (comp_ids.count(comp_id[idx]))
574 __contradiction_error(
575 "sequence",
576 "tried to set indices two indices as equal and "
577 "different");
578 comp_ids.insert(comp_id[idx]);
579 }
580
581 // Generates set of already defined values.
582 std::set<T> defined_values;
583 for (int idx : distinct)
584 if (val_range[idx].first == val_range[idx].second)
585 defined_values.insert(val_range[idx].first);
586
587 // We have values in [l, r], but those in defined_values are
588 // taken. We generate our numbers in the range [0, lim) with
589 // lim = (r-l+1)-(defined_values.size()), and then map them to
590 // the correct range.
591 // We will run (distinct.size()-defined_values.size()) steps of
592 // Fisherâ€“Yates, using a map to store a virtual sequence that starts
593 // with the a[i] = i.
594 int lim = (value_r - value_l + 1) - int(defined_values.size());
595 std::map<T, T> virtual_sequence;
596 std::vector<T> initial_gen;
597 for (int i = 0; i < distinct.size() - int(defined_values.size());
598 i++) {
599 T j = next<T>(i, lim - 1);
600 T vj = virtual_sequence.count(j) ? virtual_sequence[j] : j;
601 T vi = virtual_sequence.count(i) ? virtual_sequence[i] : T(i);
602
603 virtual_sequence[j] = vi, virtual_sequence[i] = vj;
604
605 initial_gen.push_back(virtual_sequence[i]);
606 }
607
608 // Shifts back to correct range, but there might still be values
609 // that we can not use.
610 for (T &value : initial_gen)
611 value += value_l;
612
613 // Now for every generated value that is in defined_values, we map
614 // it to [l + lim, l + lim + defined_values.size()).
615 std::vector<std::pair<T, int>> values_to_map;
616 for (int i = 0; i < initial_gen.size(); ++i)
617 if (defined_values.count(initial_gen[i])) {
618 values_to_map.emplace_back(initial_gen[i], i);
619 }
620 // We iterate through them in increasing order.
621 std::sort(values_to_map.begin(), values_to_map.end());
622 auto cur_it = defined_values.begin();
623 int cur_defined_idx = 0;
624 for (auto [val, idx] : values_to_map) {
625 while (*cur_it != val)
626 ++cur_it, ++cur_defined_idx;
627 initial_gen[idx] = value_l + lim + cur_defined_idx;
628 }
629
630 int cur = 0;
631 for (int idx : distinct) {
632 T value;
633 if (val_range[idx].first == val_range[idx].second)
634 value = val_range[idx].first; // Value was already set.
635 else
636 value = initial_gen[cur++]; // Use a generated random value.
637
638 // For every index in this component, sets it.
639 for (int i : comp[comp_id[idx]])
640 vec[i] = value;
641 }
642 }
643
644 if (values.size() > 0) {
645 // Needs to fetch the values from the value set.
646 std::vector<T> value_vec(values.begin(), values.end());
647 std::vector<T> final_vec;
648 for (int i = 0; i < size; ++i)
649 final_vec.push_back(value_vec[vec[i]]);
650 swap(vec, final_vec);
651 }
652
653 return instance(vec);
654 }
655
656 // Calls the generator until predicate is true.
657 template <typename PRED>
658 instance gen_until(PRED predicate, int max_tries,
659 bool random_default = false) {
660 for (int i = 0; i < max_tries; ++i) {
661 instance inst = gen();
662 if (predicate(inst))
663 return inst;
664 }
665 if (random_default)
666 return gen();
667 else
668 throw __error("could not generate instance matching predicate");
669 }

690// Chooses k values from the sequence, as in a subsequence of size k.
691template <typename INST> INST choose(int k, const INST &inst) {
692 tgen_ensure(0 < k and k <= inst.vec.size(),
693 "number of elements to choose must be valid");
694 std::vector<typename INST::value_type> new_vec;
695 int need = k;
