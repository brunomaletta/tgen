Examples

// Generates 100 distinct integers.
std::cout <<
tgen::sequence<int>(100, 1, 100).distinct().gen() << std::endl;
// Generates 10 sorted small powers of 2.
std::set<int> powers = {1, 2, 4, 8};
std::cout <<
tgen::sequence<int>(10, powers).gen().sort() << std::endl;
